--funkcje analityczne

select 
	department_id,
	sum(salary)
from employees
where
	department_id in (10,20,30)
group by rollup(department_id)


select 
	department_id,
	max(salary)
from employees
where
	department_id in (10,20,30)
group by rollup(department_id)

select 
	department_id,
	job_id,
	sum(salary) sum
from employees
	where department_id in (10,20,30)
group by 
	rollup(department_id, job_id)
	
	
-- 1, 2 -> 1, 12
-- 1, 2, 3 -> 1, 12, 123
-- 1, 2, 3, 4 -> 1, 12, 123, 1234
	
	
select 
	department_id,
	sum(salary) sum
from employees
where department_id in (10,20,30)
group by cube(department_id)
	

select 
	department_id,
	job_id,
	sum(salary) sum
from employees
where department_id in (10,20,30)
group by 
	cube(department_id, job_id)
	
	
select 
	department_id,
	job_id,
	sum(salary) sum
from employees
where department_id in (10,20,30)
or department_id is null
group by 
	cube(department_id, job_id)
		
	
select 
	department_id,
	job_id,
	sum(salary) sum,
	grouping(department_id) grup_dept,
	grouping(job_id) grup_job
from employees
where department_id in (10,20,30)
or department_id is null
group by 
	cube(department_id, job_id)	
	

select 
	department_id,
	job_id,
	sum(salary) sum,
	grouping(department_id) grup_dept,
	grouping(job_id) grup_job
from employees
where department_id in (10,20,30)
or department_id is null
group by 
	cube(department_id, job_id)	
having
	grouping(department_id) = 1
	and grouping(job_id) = 1
	
	
select 
	department_id,
	job_id,
	sum(salary) sum
from employees
where department_id in (10,20,30)
group by 
	grouping sets((department_id), (department_id, job_id), (job_id))
	
	
explain
select 
	department_id,
	job_id,
	sum(salary) sum
from employees
where department_id in (10,20,30)
group by 
	grouping sets((department_id), (job_id), ())
