--Podzapytania

select 
	first_name,
	last_name,
	department_id 
from employees 
where
	department_id = 
		(select department_id 
			from departments 
		where department_name = 'Shipping')

select 
	first_name,
	last_name,
	department_id 
from employees 
where
	department_id in 
	(select department_id from departments 
		where department_name in ('Shipping','Administration'))

		
select 
first_name,
last_name,
department_id
from employees
where salary between (select min_salary from jobs
						where job_title = 'Programmer')
			and (select max_salary from jobs
						where job_title = 'Stock Clerk')


select 
	department_id,
	sum(salary)
from employees
group by department_id
having sum(salary) < (select sum(salary) from employees
				where department_id = 20)

	salary,
	department_id
from employees e
where salary > (select avg(salary) from employees
				where department_id = e.department_id)

select 
	first_name,
	last_name,
	salary + (salary * coalesce(commission_pct,0)) full_salary
from employees e
where salary + (salary * coalesce(commission_pct,0)) > 5000

--table derrived
select
	t1.*
	from (
			select 
				first_name,
				last_name,
				salary + (salary * coalesce(commission_pct,0)) full_salary
			from employees e ) t1
where t1.full_salary > 5000
