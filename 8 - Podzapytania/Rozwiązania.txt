/*
1.
Wyświetl nazwiska i stanowiska pracowników pracujących 
w tym samym dziale co pracownik o nazwisku Miller (5 wierszy)
*/

select
	e.first_name ,
	e.last_name ,
	j.job_title 
	from employees e join jobs j on e.job_id = j.job_id 
where
	e.department_id = (select department_id from employees
						where last_name = 'Miller')	

/*
2.Wyświetl następujące dane o najdłużej zatrudnionym pracowniku na stanowisku 
FI_ACCOUNT (1 wiersz)
*/
select
	employee_id,
	first_name,
	last_name,
	job_id,
	hire_date
	from employees
where
	hire_date = (select min(hire_date) from employees
					where job_id = 'FI_ACCOUNT')
	and job_id = 'FI_ACCOUNT'				

select
	employee_id,
	first_name,
	last_name,
	job_id,
	hire_date
	from employees
where
	(hire_date, job_id) in (select min(hire_date), job_id from employees
							where job_id = 'FI_ACCOUNT'
							group by job_id)

/*
3.
Zobacz najdłużej pracujących pracowników każdego działu. (12 wierszy)
*/

select
	last_name,
	hire_date,
	department_id,
	(select department_name from departments where department_id = e.department_id ) dn
	from employees e
where
	hire_date = (select min(hire_date) from employees
					where department_id = e.department_id )

/*
4. Wyświetlaj te nazwy stanowisk, w których nie zatrudniono nikogo z działu
MARKETING. Wykorzystaj podzapytanie w podzapytaniu (17 wierszy)
*/
	
select
	*
	from jobs
where
	job_id in (select job_id from employees
				where department_id not in (select department_id from departments
												where department_name = 'Marketing'))
												
												
	
/*
5. Wyświetl informacje o pracownikach zarabiających więcej niż średnia pensja dla ich
etatu(job_id) (43 wiersze)
*/

select
	e.first_name ,
	e.last_name ,
	e.salary ,
	j.job_title ,
	(select avg(salary) from employees where job_id = e.job_id) avg_jobs_salary
	from employees e join jobs j on e.job_id = j.job_id 
where
	salary > (select avg(salary) from employees
				where job_id = e.job_id )


/*
6. Wyświetl nazwiska poniższe dane pracowników związanych ze sprzedażą i mających
innego szefa niż John Singh – wykorzystaj w zapytaniu strukturę „derrived column”
(29 wierszy)
*/

select 
	e.last_name,
	e.email,
	e.department_id,
	(select first_name || ' ' || last_name from employees where employee_id = e.manager_id) Boss
	from employees e 
where 
	job_id in 
				(select 
					job_id
					from jobs j 
					where lower(job_title) like '%sales%')					
	and manager_id <> 
				(select employee_id from employees						
					where first_name = 'John' 
					and last_name = 'Singh')
/*
7.
Użyj skorelowanego podzapytania, aby wyświetlić informacje o pracowniku 
niezatrudnionym w żadnym dziale 
(użyj właściwej formy operatora EXISTS) (1 wiersz)
*/

explain
select
	first_name,
	last_name,
	department_id
	from employees e
where
	not exists (select 1 from employees 
					where department_id = e.department_id )
					
/*
8. Wyświetl nazwę działu, który płaci Twoim pracownikom najwięcej 
pieniędzy miesięcznie(1 wiersz)
*/

select
	d.department_name 
	from employees e join departments d on e.department_id = d.department_id
	group by
	d.department_name
having
	sum(salary) = (
					select
						max(t1.suma)
						from (
							select
								sum(salary) suma
								from employees e 
							group by
								department_id ) t1)
