/*
1.
Wyświetl nazwiska i stanowiska pracowników pracujących 
w tym samym dziale co pracownik o nazwisku Miller (5 wierszy)
*/

select
	e.first_name ,
	e.last_name ,
	j.job_title 
	from employees e join jobs j on e.job_id = j.job_id 
where
	e.department_id = (select department_id from employees
						where last_name = 'Miller')	

/*
2.Wyświetl następujące dane o najdłużej zatrudnionym pracowniku na stanowisku 
FI_ACCOUNT (1 wiersz)
*/
select
	employee_id,
	first_name,
	last_name,
	job_id,
	hire_date
	from employees
where
	hire_date = (select min(hire_date) from employees
					where job_id = 'FI_ACCOUNT')
	and job_id = 'FI_ACCOUNT'				

select
	employee_id,
	first_name,
	last_name,
	job_id,
	hire_date
	from employees
where
	(hire_date, job_id) in (select min(hire_date), job_id from employees
							where job_id = 'FI_ACCOUNT'
							group by job_id)
