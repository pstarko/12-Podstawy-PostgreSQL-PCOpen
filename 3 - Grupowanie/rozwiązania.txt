/*
1.
Wyświetl najniższą i najwyższą pensję w firmie. Wyświetl informacje o 
różnicy między najlepiej i najgorzej opłacanymi pracownikami (1 wiersz).
*/


select
	max(salary) maxi,
	min(salary) mini,
	max(salary) - min(salary) diff
	from employees

/*
2.
Wyświetl średnie wynagrodzenie dla wszystkich job_id. Wyklucz zespoły 10,20 i 50. 
Sortuj wyniki według malejącej średniej płacy (13 wierszy).
*/


select
	job_id,
	avg(salary) srednia
	from employees
where
	department_id not in (10,20,50)
group by
	job_id
order by srednia desc
	
/*
	3. Wyświetl liczbę IT_PROG zatrudnionych w firmie (1 wiersz)
*/
	
select
	count(employee_id)
	from employees
where
	job_id = 'IT_PROG'


/*
4.
Znajdź łączne miesięczne wynagrodzenie dla każdego działu. 
Nie zapomnij o dodatkowych zarobkach. W wyniku usuń wartość null 
i dane posortuj w taki sposób by zespół 110 był pierwszy, 
a reszta już zgodnie z wartością indeksu(11 wierszy)
*/


select
	department_id,
	sum(salary + (salary * coalesce (commission_pct,0))) srednia
	from employees
where
	department_id is not null
group by
	department_id
order by 
	case when department_id = 110 then 4
	else department_id
	end

/*
6. Sprawdź, ilu mamy pracowników z literą O i R w nazwisku. Napisz zapytanie odporne
na wielkość liter w tabeli (1 wiersz)
*/


select 
	count(*) counter
	from employees
where
	lower(last_name) like '%o%' and
	lower(last_name) like '%r%'

select 
	count(*) counter
	from employees
where
	upper(last_name) like '%O%' and
	upper(last_name) like '%R%'


/*
7. Czy istnieje dział, który będzie miesięcznie zarabiał ponad 52673 (salary), gdy otrzyma
10 razy 10% podwyżki? Pokaż identyfikator tego działu i kwotę zaokrągloną do liczb
całkowitych w górę. (7 wierszy)
*/


select 
	department_id,
	ceiling(sum(salary) * power(1.1, 10)) result
	from employees
group by 
	department_id
having
	sum(salary) * power(1.1, 10) > 52673

/*
8.
Pokaż maksymalne i minimalne wynagrodzenie dla pracowników zatrudnionych po 2015r. 
i posiadających szefa o identyfikatorze 100 lub 130 (1 wiersz)
*/


select 
	max(salary),
	min(salary)
	from employees
where
	hire_date > '2015-12-31'
	and manager_id in (100, 130)


/*
9.
Wyświetl numery zespołów wraz z liczbą płac dodatkowych w każdym dziale. 
Sortuj wynik po malejącym id_działu (11 wierszy).
*/


select 
	department_id ,
	count(commission_pct)
	from employees
where
	department_id is not null
group by
	department_id
order by 
	department_id desc

/*
11.
Zbuduj zapytanie, które wyświetla średnie i maksymalne wynagrodzenie podstawowe 
IT_PROG i SA_REP w poszczególnych zespołach. 
Zaokrąglij pensję do liczb całkowitych. Posortuj wynik 
zapytania według identyfikatorów działów i identyfikatora job_id (3 wiersze).
*/


select 
	department_id,
	job_id,
	round(avg(salary)) srednia,
	round(max(salary)) maksimum
	from employees
where
	job_id in ('IT_PROG', 'SA_REP')
group by
	department_id,
	job_id
order by 
	department_id, job_id

	
/*
12. Utwórz zapytanie, które wyświetla liczbę pracowników zatrudnionych cały 2016 roku i 
w drugiej połowie 2017 roku. Posortuj wyniki według roku zatrudnienia (15 wierszy).
*/
	
	
select
	date_part('year', hire_Date),
	date_part('month', hire_Date),
	count(employee_id)
	from employees
where
	date_part('year', hire_Date) = 2016
	or (date_part('year', hire_Date) = 2017 and date_part('month', hire_Date) > 6)
group by
	date_part('year', hire_Date),
	date_part('month', hire_Date)
order by
	date_part('year', hire_Date)
