--Grupowanie

select
	min(salary) min,
	max(salary) max,
	avg(salary) avg,
	sum(salary) sum,
	count(*) count
from employees

select
	count(1) count,
	count('dzik jest dziki'),
	count(commission_pct)
from employees


select
	department_id,
	sum(salary) sum
	from employees
group by 
   department_id

   
select
	department_id,
	job_id,
	max(salary) sum
	from employees
group by 
   department_id,
   job_id
order by 1
   

select
	department_id,
	sum(salary) sum
	from employees
where
	department_id in (10,20,30)
group by 
   department_id

	department_id,
